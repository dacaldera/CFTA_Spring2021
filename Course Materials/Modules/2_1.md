## Cat Toggle 
```
PImage cat;
boolean toggle = false;


void setup(){
  size(600,600);
  cat = loadImage("cat.jpg");
}

void draw(){
  //background(255);
  
  if(toggle == true){
    image(cat, mouseX-(cat.width/2),mouseY-(cat.height/2));
  }
  
}


void mousePressed(){
  if(toggle == false){
   toggle = true; 
  } else if(toggle == true){
   toggle = false; 
  }
}
```
